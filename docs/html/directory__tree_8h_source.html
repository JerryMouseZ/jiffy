442  void load(const std::string &path,
443  const std::string &backing_path,
444  const std::shared_ptr<storage::storage_management_ops> &storage,
445  const std::shared_ptr<block_allocator> &allocator) override {
446  std::unique_lock<std::shared_mutex> lock(mtx_);
447 
448  auto auto_scale = !dstatus_.is_static_provisioned();
449  auto num_blocks = dstatus_.data_blocks().size();
450  auto chain_length = dstatus_.chain_length();
451  std::size_t slots_per_block = storage::block::SLOT_MAX / num_blocks;
452  for (std::size_t i = 0; i < num_blocks; ++i) {
453  auto slot_begin = static_cast<int32_t>(i * slots_per_block);
454  auto slot_end =
455  i == (num_blocks - 1) ? storage::block::SLOT_MAX : static_cast<int32_t>((i + 1) * slots_per_block - 1);
456  replica_chain chain(allocator->allocate(chain_length, {}),
457  slot_begin,
458  slot_end,
459  chain_status::stable,
460  storage_mode::in_memory);
461  assert(chain.block_names.size() == chain_length);
462  using namespace storage;
463  if (chain_length == 1) {
464  std::string block_backing_path = backing_path;
465  utils::directory_utils::push_path_element(block_backing_path, chain.slot_range_string());
466  storage->setup_block(chain.block_names[0],
467  path,
468  slot_begin,
469  slot_end,
470  chain.block_names,
471  auto_scale,
472  chain_role::singleton,
473  "nil");
474  storage->load(chain.block_names[0], block_backing_path);
475  dstatus_.mark_loaded(i, chain.block_names);
476  } else {
477  std::string block_backing_path = backing_path;
478  utils::directory_utils::push_path_element(block_backing_path, chain.slot_range_string());
479  for (std::size_t j = 0; j < chain_length; ++j) {
480  std::string block_name = chain.block_names[j];
481  std::string next_block_name = (j == chain_length - 1) ? "nil" : chain.block_names[j + 1];
482  int32_t role = (j == 0) ? chain_role::head : (j == chain_length - 1) ? chain_role::tail : chain_role::mid;
483  storage->setup_block(block_name,
484  path,
485  slot_begin,
486  slot_end,
487  chain.block_names,
488  auto_scale,
489  role,
490  next_block_name);
491  storage->load(block_name, block_backing_path);
492  }
493  dstatus_.mark_loaded(i, chain.block_names);
494  }
495  }

509  bool handle_lease_expiry(std::vector<std::string> &cleared_blocks,
510  std::shared_ptr<storage::storage_management_ops> storage) {
511  std::unique_lock<std::shared_mutex> lock(mtx_);
512  if (!dstatus_.is_pinned()) {
513  using namespace utils;
514  LOG(log_level::info) << "Clearing storage for " << name();
515  if (dstatus_.is_mapped()) {
516  for (const auto &block: dstatus_.data_blocks()) {
517  for (size_t i = 0; i < dstatus_.chain_length(); i++) {
518  if (i == dstatus_.chain_length() - 1) {

523  } else {
524  storage->reset(block.block_names[i]);
525  }
526  cleared_blocks.push_back(block.block_names[i]);
527  }
528  }
529  return false; // Clear the blocks, but don't delete the path
530  } else {
531  for (const auto &block: dstatus_.data_blocks()) {
532  for (const auto &block_name: block.block_names) {
533  storage->reset(block_name);
534  cleared_blocks.push_back(block_name);
535  }
536  }
537  }
538  return true; // Clear the blocks and delete the path
539  }
540  return false; // Don't clear the blocks or delete the path

551  export_ctx setup_add_block(std::shared_ptr<storage::storage_management_ops> storage,
552  const std::shared_ptr<block_allocator> &allocator,
553  const std::string &path) {
554  using namespace storage;
555  if (dstatus_.data_blocks().size() == block::SLOT_MAX)
556  throw directory_ops_exception("Cannot expand capacity beyond " + std::to_string(block::SLOT_MAX) + " blocks");
557 
558  std::unique_lock<std::shared_mutex> lock(mtx_);
559  // Get the block with largest size
560  std::vector<std::future<std::size_t>> futures;
561  for (const auto &block: dstatus_.data_blocks()) {
562  futures.push_back(std::async([&]() -> std::size_t { return storage->storage_size(block.block_names.back()); }));
563  }
564  size_t i = 0;
565  size_t max_size = 0;
566  size_t max_pos = 0;
567  for (auto &future: futures) {
568  size_t sz = future.get();
569  auto cstatus = dstatus_.get_data_block_status(i);
570  if (sz > max_size && cstatus != chain_status::exporting && cstatus != chain_status::importing
571  && dstatus_.num_slots(i) != 1) {
572  max_size = sz;
573  max_pos = i;
574  }
575  i++;
576  }
577  dstatus_.set_data_block_status(max_pos, chain_status::exporting);
578 
579  // Split the block's slot range in two
580  auto from_chain = dstatus_.data_blocks().at(max_pos);
581  auto slot_begin = from_chain.slot_range.first;
582  auto slot_end = from_chain.slot_range.second;
583  auto slot_mid = (slot_end + slot_begin) / 2; // TODO: We can get a better split...
584 
585  // Allocate the new chain
586  replica_chain to_chain(allocator->allocate(dstatus_.chain_length(), {}),
587  slot_mid + 1,
588  slot_end,
589  chain_status::stable,
590  storage_mode::in_memory);
591  assert(to_chain.block_names.size() == chain_length);
592 
593  // Set old chain to exporting and new chain to importing
594  if (dstatus_.chain_length() == 1) {
595  storage->setup_and_set_importing(to_chain.block_names[0],
596  path,
597  slot_mid + 1,
598  slot_end,
599  to_chain.block_names,
600  chain_role::singleton,
601  "nil");
602  storage->set_exporting(from_chain.block_names[0], to_chain.block_names, slot_mid + 1, slot_end);
603  } else {
604  for (std::size_t j = 0; j < dstatus_.chain_length(); ++j) {
605  std::string block_name = to_chain.block_names[j];
606  std::string next_block_name = (j == dstatus_.chain_length() - 1) ? "nil" : to_chain.block_names[j + 1];
607  int32_t
608  role =
609  (j == 0) ? chain_role::head : (j == dstatus_.chain_length() - 1) ? chain_role::tail : chain_role::mid;
610  storage->setup_and_set_importing(block_name,
611  path,
612  slot_mid + 1,
613  slot_end,
614  to_chain.block_names,
615  role,
616  next_block_name);
617  }
618  for (std::size_t j = 0; j < dstatus_.chain_length(); ++j) {
619  storage->set_exporting(from_chain.block_names[j], to_chain.block_names, slot_mid + 1, slot_end);
620  }
621  }
622 
623  adding_.push_back(to_chain);
624  return export_ctx{from_chain, to_chain};
625  }

718  void finalize_slot_range_split(std::shared_ptr<storage::storage_management_ops> storage, const export_ctx &ctx) {
719  std::unique_lock<std::shared_mutex> lock(mtx_);
720  auto slot_begin = ctx.from_block.slot_begin();
721  auto slot_end = ctx.from_block.slot_end();
722  auto slot_mid = (slot_end + slot_begin) / 2;
723 
724  auto from_idx = dstatus_.find_replica_chain(ctx.from_block);
725  dstatus_.update_data_block_slots(from_idx, slot_begin, slot_mid);
726  dstatus_.set_data_block_status(from_idx, chain_status::stable);
727  dstatus_.add_data_block(ctx.to_block, from_idx + 1);
728  auto it = std::find(adding_.begin(), adding_.end(), ctx.to_block);
729  if (it == adding_.end()) {
730  throw std::logic_error("Cannot find to_block in adding list");
731  }
732  adding_.erase(it);
733  for (std::size_t j = 0; j < dstatus_.chain_length(); ++j) {
734  storage->set_regular(ctx.from_block.block_names[j], slot_begin, slot_mid);
735  storage->set_regular(ctx.to_block.block_names[j], slot_mid + 1, slot_end);
736  }
737  using namespace utils;
738  LOG(log_level::info) << "Updated file data_status: " << dstatus_.to_string();

749  export_ctx setup_slot_range_merge(std::shared_ptr<storage::storage_management_ops> storage,
750  int32_t slot_begin,
751  int32_t slot_end) {
752  using namespace storage;
753  if (dstatus_.data_blocks().size() == 1 || slot_end == block::SLOT_MAX) {
754  throw directory_ops_exception("Cannot find a merge partner");
755  }
756  std::unique_lock<std::shared_mutex> lock(mtx_);
757  size_t block_idx = 0;
758  for (const auto &block: dstatus_.data_blocks()) {
759  if (block.slot_begin() == slot_begin && block.slot_end() == slot_end) {
760  break;
761  }
762  block_idx++;
763  }
764  if (block_idx == dstatus_.data_blocks().size()) {
765  throw directory_ops_exception(
766  "No block with slot range " + std::to_string(slot_begin) + "-" + std::to_string(slot_end));
767  }
768  auto cstatus = dstatus_.get_data_block_status(block_idx);
769  if (cstatus == chain_status::exporting || cstatus == chain_status::importing) {
770  throw directory_ops_exception("Block already involved in re-partitioning");
771  }
772 
773  auto from_chain = dstatus_.data_blocks().at(block_idx);
774  // Always merge with the right neighbor; we can find the better neighbor, but we just don't
775  auto to_chain = dstatus_.data_blocks().at(block_idx + 1);
776  if (to_chain.status == chain_status::exporting) {
777  throw directory_ops_exception("Cannot find a merge partner");
778  }
779 
780  dstatus_.set_data_block_status(block_idx, chain_status::exporting);
781  dstatus_.set_data_block_status(block_idx + 1, chain_status::importing);
782 
783  // Set old chain to exporting and new chain to importing
784  if (dstatus_.chain_length() == 1) {
785  storage->set_importing(to_chain.block_names[0], slot_begin, slot_end);
786  storage->set_exporting(from_chain.block_names[0], to_chain.block_names, slot_begin, slot_end);
787  } else {
788  for (std::size_t j = 0; j < dstatus_.chain_length(); ++j) {
789  storage->set_importing(to_chain.block_names[j], slot_begin, slot_end);
790  }
791  for (std::size_t j = 0; j < dstatus_.chain_length(); ++j) {
792  storage->set_exporting(from_chain.block_names[j], to_chain.block_names, slot_begin, slot_end);
793  }
794  }
795  return export_ctx{from_chain, to_chain};
796  }

805  void finalize_slot_range_merge(std::shared_ptr<storage::storage_management_ops> storage,
806  const std::shared_ptr<block_allocator> &allocator,
807  const export_ctx &ctx) {
808  std::unique_lock<std::shared_mutex> lock(mtx_);
809  auto slot_begin = ctx.from_block.slot_begin();
810  auto slot_end = ctx.to_block.slot_end();
811  auto to_idx = dstatus_.find_replica_chain(ctx.to_block);
812  dstatus_.update_data_block_slots(to_idx, slot_begin, slot_end);
813  dstatus_.set_data_block_status(to_idx, chain_status::stable);
814  auto from_idx = dstatus_.find_replica_chain(ctx.from_block);
815  dstatus_.remove_data_block(from_idx);
816  for (std::size_t j = 0; j < dstatus_.chain_length(); ++j) {
817  storage->reset(ctx.from_block.block_names[j]);
818  storage->set_regular(ctx.to_block.block_names[j], slot_begin, slot_end);
819  }

922  if (!dir->handle_lease_expiry(cleared_blocks, entry.first, storage)) {
923  cleared = false;
924  }
925  }
926  if (cleared)
927  children_.erase(ret);
928  return cleared;
929  }

932  }
