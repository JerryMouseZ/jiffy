35  static std::string stacktrace() {
36  std::ostringstream out;
37  out << "stack trace: \n";
38 
39  // storage array for stack trace address data
40  void *addrlist[MAX_FRAMES + 1];
41 
42  // retrieve current stack addresses
43  int addrlen = backtrace(addrlist, sizeof(addrlist) / sizeof(void *));
44 
45  if (addrlen == 0) {
46  out << " <empty, possibly corrupt>\n";
47  return out.str();
48  }
49 
50  // resolve addresses into strings containing "filename(function+address)",
51  // this array must be free()-ed
52  char **symbollist = backtrace_symbols(addrlist, addrlen);
53 
54  // allocate string which will be filled with the demangled function name
55  size_t funcnamesize = 256;
56  auto *funcname = (char *) malloc(funcnamesize);
57 
58  // iterate over the returned symbol lines. skip the first, it is the
59  // address of this function.
60  for (int i = 1; i < addrlen; i++) {
61  char *begin_name = nullptr, *begin_offset = nullptr, *end_offset = nullptr;
62 
63  // find parentheses and +address offset surrounding the mangled name:
64  // ./module(function+0x15c) [0x8048a6d]
65  for (char *p = symbollist[i]; *p; ++p) {
66  if (*p == '(')
67  begin_name = p;
68  else if (*p == '+')
69  begin_offset = p;
70  else if (*p == ')' && begin_offset) {
71  end_offset = p;
72  break;
73  }
74  }
75 
76  if (begin_name && begin_offset && end_offset
77  && begin_name < begin_offset) {
78  *begin_name++ = '\0';
79  *begin_offset++ = '\0';
80  *end_offset = '\0';
81 
82  // mangled name is now in [begin_name, begin_offset) and caller
83  // offset in [begin_offset, end_offset). now apply
84  // __cxa_demangle():
85 
86  int status;
87  char *ret = abi::__cxa_demangle(begin_name, funcname, &funcnamesize,
88  &status);
89  if (status == 0) {
90  funcname = ret; // use possibly realloc()-ed string
91  out << " " << symbollist[i] << ": " << funcname << "+"
92  << begin_offset;
93  } else {
94  // demangling failed. Output function name as a C function with
95  // no arguments.
96  out << " " << symbollist[i] << ": " << begin_name << "()+"
97  << begin_offset;
98  }
99 
100  char syscom[256];
101  sprintf(syscom, "addr2line %p -e %s", addrlist[i], symbollist[i]);
102  FILE *cmd = popen(syscom, "r");
103  if (cmd) {
104  char buf[256];
105  int n = fscanf(cmd, "%256s", buf);
106  pclose(cmd);
107  if (n != EOF)
108  out << "(" << buf << ")\n";
109  } else {
110  out << "\n";
111  }
112  } else {
113  // couldn't parse the line? print the whole line.
114  out << " " << symbollist[i] << "\n";
115  }
116  }
117 
118  free(funcname);
119  free(symbollist);
120  return out.str();
121  }
